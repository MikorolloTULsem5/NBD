1   package nbd.gV.repositories;
2   
3   import jakarta.persistence.EntityManager;
4   import jakarta.persistence.LockModeType;
5   import jakarta.persistence.LockTimeoutException;
6   import jakarta.persistence.PersistenceException;
7   import jakarta.persistence.PessimisticLockException;
8   import jakarta.persistence.TransactionRequiredException;
9   import jakarta.persistence.criteria.CriteriaBuilder;
10  import jakarta.persistence.criteria.CriteriaQuery;
11  import jakarta.persistence.criteria.Root;
12  import nbd.gV.clients.Client;
13  import nbd.gV.clients.ClientType;
14  import nbd.gV.clients.Client_;
15  import nbd.gV.exceptions.JakartaException;
16  
17  import java.util.List;
18  import java.util.UUID;
19  
20  public class ClientRepository extends Repository<Client> {
21  
22      public ClientRepository(String unitName) {
23          super(unitName);
24      }
25  
26      private ClientType returnClientTypeFromDB(ClientType clientType) {
27          CriteriaBuilder cb = getEntityManager().getCriteriaBuilder();
28          CriteriaQuery<Client> query = cb.createQuery(Client.class);
29          Root<Client> clientRoot = query.from(Client.class);
30          query.select(clientRoot).where(cb.equal(clientRoot.get(Client_.CLIENT_TYPE), clientType));
31          List<Client> result = find(query);
32  
33          return result.isEmpty() ? null : result.get(0).getClientType();
34      }
35  
36      @Override
37      public void create(Client newClient) {
38          EntityManager entityManager = getEntityManager();
39          try {
40              ClientType clientTypeFromDB = returnClientTypeFromDB(newClient.getClientType());
41  
42              entityManager.getTransaction().begin();
43              if (clientTypeFromDB != null) {
44                  newClient.setClientType(clientTypeFromDB);
45              }
46              entityManager.persist(newClient);
47              entityManager.getTransaction().commit();
48          } catch (PersistenceException | IllegalArgumentException | IllegalStateException
49                   | NullPointerException exception) {
50              if (entityManager.getTransaction().isActive()) {
51                  entityManager.getTransaction().rollback();
52              }
53              throw new JakartaException(exception.getMessage());
54          }
55      }
56  
57      @Override
58      public Client findByUUID(UUID identifier) {
59          Client returnClient;
60          try{
61              getEntityManager().getTransaction().begin();
62              returnClient = getEntityManager().find(Client.class, identifier, LockModeType.PESSIMISTIC_READ);
63              getEntityManager().getTransaction().commit();
64          }catch (IllegalArgumentException | TransactionRequiredException | PessimisticLockException
65                  | LockTimeoutException exception) {
66              getEntityManager().getTransaction().rollback();
67              throw new JakartaException(exception.getMessage());
68          }
69          return returnClient;
70      }
71  
72      @Override
73      public List<Client> findAll() {
74          CriteriaQuery<Client> findAllClients = getEntityManager().getCriteriaBuilder().createQuery(Client.class);
75          Root<Client> courtClient = findAllClients.from(Client.class);
76          findAllClients.select(courtClient);
77          return find(findAllClients);
78      }
79  }
