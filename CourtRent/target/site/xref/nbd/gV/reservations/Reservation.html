4   import jakarta.validation.constraints.NotNull;

10  import java.time.Duration;
11  import java.time.LocalDateTime;
12  import java.time.format.DateTimeFormatter;
13  import java.util.Formatter;
14  import java.util.Locale;
15  import java.util.UUID;
16  
17  @Entity
18  @Table(name = "reservation")
19  @Access(value = AccessType.FIELD)
20  public class Reservation {
21      @Id
22      private UUID id;
23      @NotNull
24      @ManyToOne
25      @JoinColumn
26      private Client client;
27      @NotNull
28      @ManyToOne
29      @JoinColumn
30      private Court court;
31      @NotNull
32      private LocalDateTime beginTime;
33      private LocalDateTime endTime = null;
34      private double reservationCost;
35  
36      public Reservation(UUID id, Client client, Court court, LocalDateTime beginTime) {
37          if (id == null || client == null || court == null)
38              throw new MainException("Niepoprawny parametr przy tworzeniu obiektu rezerwacji!");
39  
40          this.id = id;
41          this.client = client;
42          this.court = court;
43          this.beginTime = (beginTime == null) ? LocalDateTime.now() : beginTime;
44          court.setRented(true);
45      }
46  
47      public Reservation() {
48      }
49  
50      public UUID getId() {
51          return id;
52      }
53  
54      public Client getClient() {
55          return client;
56      }
57  
58      public Court getCourt() {
59          return court;
60      }
61  
62      public LocalDateTime getBeginTime() {
63          return beginTime;
64      }
65  
66      public LocalDateTime getEndTime() {
67          return endTime;
68      }
69  
70      public double getReservationCost() {
71          return reservationCost;
72      }
73  
74      public int getReservationHours() {
75          int hours = 0;
76  
77          if (endTime != null) {
78              long duration = Duration.between(beginTime, endTime).getSeconds();
79              int hoursDur = (int) (duration / 3600);
80              int minutesDur = (int) ((duration / 60) % 60);
81  
82              if (!(hoursDur == 0 && minutesDur == 0)) {
83                  hours = (minutesDur == 0) ? hoursDur : (hoursDur + 1);
84              }
85          }
86  
87          return hours;
88      }
89  
90      public void endReservation(LocalDateTime endingDate) {
91          if (endTime == null) {
92              endTime = (endingDate == null) ? LocalDateTime.now() : endingDate;
93              if (Duration.between(beginTime, endTime).isNegative()) {
94                  endTime = beginTime;
95              }
96  
97              court.setRented(false);
98  
99              if (getReservationHours() <= client.getClientMaxHours()) {
100                 reservationCost = getReservationHours() * court.getBaseCost() -
101                         client.applyDiscount(getReservationHours() * court.getBaseCost());
102             } else {
103                 reservationCost = court.getBaseCost() *
104                         (client.getClientMaxHours() + (getReservationHours() - client.getClientMaxHours()) * 1.5) -
105                         client.applyDiscount(client.getClientMaxHours() * court.getBaseCost());
106             }
107         } else {
108             throw new ReservationException("Ta rezerwacja juz sie zakonczyla i nie mozna zmienic jej daty!");
109         }
110     }
111 
112     public String getReservationInfo() {
113         return new Formatter(Locale.GERMAN).format("Rezerwacja nr %s przez '%s' boiska: '%s', od godziny [%s]%s%n", id,
114                 client.getClientInfo().replace("\n", ""),
115                 court.getCourtInfo().replace("\n", ""),
116                 beginTime.format(DateTimeFormatter.ofPattern("dd.MM.yyyy, HH:mm")),
117                 (endTime == null) ? "." : (" do godziny [%s].".formatted(
118                         endTime.format(DateTimeFormatter.ofPattern("dd.MM.yyyy, HH:mm"))))).toString();
119     }
