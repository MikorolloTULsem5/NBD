1   package nbd.gV.clients;
2   
3   import jakarta.persistence.criteria.CriteriaBuilder;
4   import jakarta.persistence.criteria.CriteriaQuery;
5   import jakarta.persistence.criteria.Root;
6   import nbd.gV.exceptions.ClientException;
7   import nbd.gV.exceptions.CourtException;
8   import nbd.gV.exceptions.JakartaException;
9   import nbd.gV.exceptions.MainException;
10  import nbd.gV.repositories.ClientRepository;
11  
12  import java.util.List;
13  import java.util.UUID;
14  
15  public class ClientManager {
16  
17      private final ClientRepository clientRepository;
18  
19      public ClientManager(String unitName) {
20          this.clientRepository = new ClientRepository(unitName);
21      }
22  
23      public ClientManager() {
24          this.clientRepository = new ClientRepository("default");
25      }
26  
27      public Client registerClient(String firstName, String lastName, String personalID, ClientType clientType) {
28          Client newClient = new Client(firstName, lastName, personalID, clientType);
29          try {
30              clientRepository.create(newClient);
31          } catch (JakartaException exception) {
32              throw new ClientException("Nie udalo sie zarejestrowac klienta w bazie!");
33          }
34          return newClient;
35      }
36  
37      public void unregisterClient(Client client) {
38          if (client == null) {
39              throw new MainException("Nie mozna wyrejestrowac nieistniejacego klienta!");
40          }
41          try {
42              client.setArchive(true);
43              clientRepository.update(client);
44          } catch (JakartaException exception) {
45              client.setArchive(false);
46              throw new ClientException("Nie udalo sie wyrejestrowac podanego boiska.");
47          }
48      }
49  
50      public Client getClient(UUID clientID) {
51          try {
52              return clientRepository.findByUUID(clientID);
53          } catch (JakartaException exception) {
54              throw new ClientException("Blad transakcji.");
55          }
56      }
57  
58      public List<Client> getAllClients() {
59          try {
60              return clientRepository.findAll();
61          } catch (JakartaException exception) {
62              throw new CourtException("Nie udalo sie uzyskac clientow.");
63          }
64      }
65  
66      public Client findCourtByPersonalId(String personalId) {
67          CriteriaBuilder cb = clientRepository.getEntityManager().getCriteriaBuilder();
68          CriteriaQuery<Client> query = cb.createQuery(Client.class);
69          Root<Client> clientRoot = query.from(Client.class);
70          query.select(clientRoot).where(cb.equal(clientRoot.get(Client_.PERSONAL_ID), personalId));
71          List<Client> result = clientRepository.find(query);
72          return result.isEmpty() ? null : result.get(0);
73      }
74  }
