109     @Test
110     void testFindingRecordsInDB() {
111         Client client1 = new Client("John", "Smith", "12345678911", new Normal());
112         clientRepository.create(client1);
113         Client client2 = new Client("Adam", "Red", "12345678912", new Normal());
114         clientRepository.create(client2);
115         Client client3 = new Client("Adam", "Black", "12345678913", new Normal());
116         clientRepository.create(client3);
117         Client client4 = new Client("Jan", "Smith", "12345678914", new Normal());
118         clientRepository.create(client4);
119 
120         //Tworzenie zapytania o klientow o imieniu Adam
121         CriteriaBuilder cb = clientRepository.getEntityManager().getCriteriaBuilder();
122         CriteriaQuery<Client> query = cb.createQuery(Client.class);
123         Root<Client> clientRoot = query.from(Client.class);
124         query.select(clientRoot).where(cb.equal(clientRoot.get(Client_.FIRST_NAME), "Adam"));
125 
126         List<Client> resultFirstName = clientRepository.find(query);
127         assertEquals(2, resultFirstName.size());
128         assertEquals(client2, resultFirstName.get(0));
129         assertEquals(client3, resultFirstName.get(1));
130 
131         //Tworzenie zapytania o boiska o nazwisku Smith
132         query.select(clientRoot).where(cb.equal(clientRoot.get(Client_.LAST_NAME), "Smith"));
133 
134         List<Client> resultLastName = clientRepository.find(query);
135         assertEquals(2, resultLastName.size());
136         assertEquals(client1, resultLastName.get(0));
137         assertEquals(client4, resultLastName.get(1));
138 
139         //Tworzenie zapytania o klienta o nazwisku Yellow - taki klient nie istnieje
140         query.select(clientRoot).where(cb.equal(clientRoot.get(Client_.LAST_NAME), "Yellow"));
141         assertTrue(clientRepository.find(query).isEmpty());
142 
143         //Pobranie wszystkich boisk z bazy
144         List<Client> resultAll = clientRepository.findAll();
145         assertEquals(4, resultAll.size());
146         assertEquals(client1, resultAll.get(0));
147         assertEquals(client2, resultAll.get(1));
148         assertEquals(client3, resultAll.get(2));
149         assertEquals(client4, resultAll.get(3));
150     }
