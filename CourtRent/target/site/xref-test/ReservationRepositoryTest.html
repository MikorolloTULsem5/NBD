167     @Test
168     void testFindingRecordsInDB() {
169         ReservationRepository reservationRepository = new ReservationRepository("test");
170         assertNotNull(reservationRepository);
171 
172         Reservation reservation1 = reservationRepository.create(testClient1,testCourt1,testTimeStart);
173         Reservation reservation2 = reservationRepository.create(testClient1,testCourt2,testTimeStart);
174         Reservation reservation3 = reservationRepository.create(testClient2,testCourt3,testTimeStart);
175 
176         CriteriaBuilder cb = reservationRepository.getEntityManager().getCriteriaBuilder();
177         CriteriaQuery<Reservation> query = cb.createQuery(Reservation.class);
178         Root<Reservation> courtRoot = query.from(Reservation.class);
179         query.select(courtRoot).where(cb.equal(courtRoot.get(Reservation_.CLIENT), testClient1));
180 
181         List<Reservation> reservations = reservationRepository.find(query);
182         assertEquals(2, reservations.size());
183         assertEquals(reservation1, reservations.get(0));
184         assertEquals(reservation2, reservations.get(1));
185 
186         query.select(courtRoot).where(cb.equal(courtRoot.get(Reservation_.CLIENT), testClient3));
187         assertTrue(reservationRepository.find(query).isEmpty());
188 
189         List<Reservation> resultAll = reservationRepository.findAll();
190         assertEquals(3, resultAll.size());
191         assertEquals(reservation1, resultAll.get(0));
192         assertEquals(reservation2, resultAll.get(1));
193         assertEquals(reservation3, resultAll.get(2));
